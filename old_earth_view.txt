  <script>
    let map3D = null;
    let geocoder = null;
    let currentLat = 0;
    let currentLng = 0;
    let orbitInterval = null;
    let currentHeading = 0;
    let currentView = 'street';

    const statusDiv = document.getElementById('status');
    const addressDisplayDiv = document.getElementById('addressDisplay');

    const viewPresets = {
      street: {
        altitude: 30,
        range: 150,
        tilt: 75,
      },
      close: {
        altitude: 20,
        range: 60,
        tilt: 78,
      },
      aerial: {
        altitude: 400,
        range: 1000,
        tilt: 60,
      }
    };

    async function init() {
      const { Geocoder } = await google.maps.importLibrary('geocoding');
      const { Map3DElement, MapMode } = await google.maps.importLibrary('maps3d');
      await google.maps.importLibrary('streetView');
      await google.maps.importLibrary('geometry');

      geocoder = new Geocoder();
      window._MapMode = MapMode;
      window._Map3DElement = Map3DElement;

      searchAddress();
    }

    async function searchAddress() {
      if (!geocoder) {
        statusDiv.innerHTML = '<div class="loading">Loading API...</div>';
        return;
      }

      const address = document.getElementById('address').value.trim();
      if (!address) {
        statusDiv.innerHTML = '<div class="error">Please enter an address</div>';
        return;
      }

      statusDiv.innerHTML = '<div class="loading">Searching...</div>';
      addressDisplayDiv.innerHTML = '';
      stopOrbit();

      try {
        const response = await geocoder.geocode({ address: address });

        if (!response.results || response.results.length === 0) {
          throw new Error('Address not found.');
        }

        const location = response.results[0];
        currentLat = location.geometry.location.lat();
        currentLng = location.geometry.location.lng();
        const formattedAddress = location.formatted_address;

        addressDisplayDiv.innerHTML = `
          <div class="address-display">
            <strong>Found:</strong><br>
            ${formattedAddress}
          </div>
        `;

        // Auto-detect front of house from nearest street
        statusDiv.innerHTML = '<div class="loading">Finding street direction...</div>';
        currentHeading = await getStreetHeading(currentLat, currentLng);
        load3DMap(currentLat, currentLng, formattedAddress);

      } catch (error) {
        statusDiv.innerHTML = `<div class="error">${error.message}</div>`;
        console.error('Search error:', error);
      }
    }

    async function getStreetHeading(lat, lng) {
      const sv = new google.maps.StreetViewService();
      const targetLatLng = new google.maps.LatLng(lat, lng);

      return new Promise((resolve) => {
        sv.getPanorama(
          {
            location: targetLatLng,
            radius: 200,
            preference: google.maps.StreetViewPreference.NEAREST,
            source: google.maps.StreetViewSource.OUTDOOR,
          },
          (data, status) => {
            if (status !== 'OK' || !data || !data.location || !data.location.latLng) {
              console.log('No street view panorama found, defaulting heading to 0');
              resolve(0);
              return;
            }

            // Heading FROM the street TO the house = front-facing direction
            const streetLatLng = data.location.latLng;
            const heading = google.maps.geometry.spherical.computeHeading(
              streetLatLng,
              targetLatLng
            );

            const normalized = (heading + 360) % 360;
            console.log(
              `Street panorama at ${streetLatLng.lat().toFixed(5)}, ${streetLatLng.lng().toFixed(5)}. ` +
              `Heading to house: ${normalized.toFixed(1)}°`
            );
            resolve(normalized);
          }
        );
      });
    }

    function load3DMap(lat, lng, address) {
      try {
        const Map3DElement = window._Map3DElement;
        const MapMode = window._MapMode;

        if (map3D && map3D.parentNode) {
          map3D.parentNode.removeChild(map3D);
          map3D = null;
        }

        const preset = viewPresets[currentView] || viewPresets.street;

        map3D = new Map3DElement({
          center: {
            lat: lat,
            lng: lng,
            altitude: preset.altitude
          },
          range: preset.range,
          tilt: preset.tilt,
          heading: currentHeading,
          mode: MapMode.SATELLITE,
        });

        document.body.append(map3D);

        statusDiv.innerHTML = `<div class="success">✓ Loaded (heading: ${currentHeading.toFixed(0)}°)</div>`;

      } catch (error) {
        statusDiv.innerHTML = `<div class="error">Error: ${error.message}</div>`;
        console.error('Map error:', error);
      }
    }

    function setView(viewName) {
      stopOrbit();
      currentView = viewName;

      document.querySelectorAll('.view-btn[data-view]').forEach(btn => {
        btn.classList.toggle('active', btn.dataset.view === viewName);
      });
      document.querySelector('[data-view="orbit"]').classList.remove('active');

      if (!map3D) return;

      const preset = viewPresets[viewName];
      if (!preset) return;

      load3DMap(currentLat, currentLng, '');
      statusDiv.innerHTML = `<div class="success">✓ View: ${viewName} (heading: ${currentHeading.toFixed(0)}°)</div>`;
    }

    function setHeading(heading) {
      currentHeading = heading;
      if (map3D) {
        map3D.heading = heading;
        statusDiv.innerHTML = `<div class="success">✓ Heading: ${heading}°</div>`;
      }
    }

    function toggleOrbit() {
      const orbitBtn = document.querySelector('[data-view="orbit"]');

      if (orbitInterval) {
        stopOrbit();
        return;
      }

      orbitBtn.classList.add('active');
      document.querySelectorAll('.view-btn:not([data-view="orbit"])').forEach(b => b.classList.remove('active'));

      orbitInterval = setInterval(() => {
        if (!map3D) return;
        currentHeading = (currentHeading + 0.3) % 360;
        map3D.heading = currentHeading;
      }, 16);
    }

    function stopOrbit() {
      if (orbitInterval) {
        clearInterval(orbitInterval);
        orbitInterval = null;
      }
      const orbitBtn = document.querySelector('[data-view="orbit"]');
      if (orbitBtn) orbitBtn.classList.remove('active');
    }

    function setAddress(address) {
      document.getElementById('address').value = address;
      searchAddress();
    }

    document.getElementById('address').addEventListener('keypress', (e) => {
      if (e.key === 'Enter') searchAddress();
    });

    init();
  </script>
